This is MFE5 web component

To install the Dell Design System Angular library, use the following command in your terminal: 
npm config set strict-ssl false
npm install @dds/angular --registry=https://artifacts.dell.com/artifactory/api/npm/dx-npm-prod

Component modules The @dds/angular package contains everything you need to use DDS Angular components in your project. In your application bootstrap, normally the app.module.ts file, import the module DDSAngularModule.

In your application bootstrap, normally the app.module.ts file, import the module DDSAngularModule. 

import {{ '{' }} NgModule{{ '}' }} from "@angular/core";
import {{ '{' }} BrowserModule{{ '}' }} from "@angular/platform-browser";
import {{ '{' }} AppRoutingModule{{ '}' }} from "./app-routing.module";
import {{ '{' }} AppComponent{{ '}' }} from "./app.component";
import {{ '{' }} DDSAngularModule{{ '}' }} from "@dds/angular";
@NgModule({{ '{' }}
 declarations: [AppComponent],
 imports: [BrowserModule, AppRoutingModule, DDSAngularModule],
 providers: [],
 bootstrap: [AppComponent]
{{ '}' }} )
export class AppModule {{ '{' }}{{ '}' }}

Standalone Components If your application is configured to use standalone components, you can still use DDS/Angular library simply importing DDSAngularModule (or each individual component module) in the @Component decorator of each module of your application. 
@Component({{ '{' }}
 selector: "my-component",
 standalone: true,
 imports: [..., DDSAngularModule]
  ...
{{ '}' }})
export class MyComponent {{ '{' }}

{{ '}' }}
